 import { BackToTop } from 'earth-ui'
 
 const BackToTopBasic = () => {
   const config = {
     icon: 'eject',
     topDistance: 10,
     timing: 'easeIn',
     background: 'rgba(73, 73, 73, 1)',
     hover: {background: 'rgba(43, 43, 43, 1)'},
     position: {
       bottom: '10%',
       right: '40px'
     }
   }
   return (
     <BackToTop {...config} />
   )
 }
