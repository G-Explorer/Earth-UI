import { Header, Button, notification } from '@webapp-suite/ui'

export const ColorfulHeader = () => {
  const colors = [
    'white',
    'blue',
    'red',
    'orange',
    'yellow',
    'green',
    'purple',
    'pink',
    'gray',
    'slate',
    'black'
  ]
  const ul = {
    lineHeight: '20px' // todo: fix button line-height caused by .markdown
  }
  const li = {
    width: '100%',
    listStyle: 'none',
    marginBottom: 20
  }
  if (window.matchMedia('(min-width:1800px)').matches) {
    ul.columnCount = 6
  }
  return (
    <div>
      <ul style={ul}>
        {colors.map(color =>
          <li style={li} key={color}>
            <Header icon={`https://cosmos-x.oss-cn-hangzhou.aliyuncs.com/${color === 'white' ? 'blue' : 'white'}-bg-webapps-ui-icon.png`} title={color} color={color}>
              <Button ghost onClick={() => notification.success('You have clicked this button!')}>BUTTON</Button>
            </Header>
          </li>
        )}
      </ul>
    </div>
  )
}

<ColorfulHeader/>
