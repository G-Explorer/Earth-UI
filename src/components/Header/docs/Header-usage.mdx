## Basic Usage

```jsx vertical
import { Header } from 'earth-ui'

const HeaderBasic = () => {
  return (
    <Header icon="/svg/app_logo_bg_blue.svg" title="Header" />
  )
}
```


## Background color

```jsx vertical
import { Header } from 'earth-ui'

const ColorfulHeader = () => {
  return (
    <Header icon="/svg/app_logo_bg_white.svg" title="Header" color="blue" />
  )
}
```
