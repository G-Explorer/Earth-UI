## 基本功能

```jsx tabs
import { Select, Option } from '@webapp-suite/ui'

const SelectBasic = () => {
  return (
    <Select width={200}>
      <Option>请选择</Option>
      <Option value="0">苹果</Option>
      <Option value="1">三星</Option>
      <Option value="2">小米</Option>
    </Select>
  )
}
```

## data 数据

```jsx tabs
import { Select, Option } from '@webapp-suite/ui'

const SelectData = () => {
  const data = [{
    id: 0,
    name: '苹果'
  }, {
    id: 1,
    name: '三星'
  }, {
    id: 2,
    name: '小米'
  }]
  return (
    <Select
      data={data}
      render={item => <Option value={item.id}>{item.name}</Option>}
      defaultOption={<Option>请选择</Option>}
      width={200}
    />
  )
}
```
