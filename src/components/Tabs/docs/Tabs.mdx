## Basic usage

```jsx vertical
import { Tabs, TabList, Tab, TabPanel } from '@webapps-ui/core-react'

const TabsBasic = () => {
  return (
    <Tabs>
      <TabList>
        <Tab>One</Tab>
        <Tab>Two</Tab>
      </TabList>
      <TabPanel>This is One!</TabPanel>
      <TabPanel>This is Two!</TabPanel>
    </Tabs>
  )
}
```

## Add or Close Tabs

```jsx vertical
import { Tabs, TabList, Tab, TabPanel, Button } from '@webapps-ui/core-react'

class TabsDynamic extends React.Component {
  constructor(props) {
    super()
    this.state = {
      tabs: [{
        id: 'A',
        name: 'Tab A'
      }, {
        id: 'B',
        name: 'Tab B'
      }]
    }
  }
  handleClose(index) {
    const tabs = this.state.tabs
    tabs.splice(index, 1)
    this.setState({ tabs, activeIndex: tabs.length - 1 })
  }
  handleAdd(e) {
    e.preventDefault()
    const tabs = this.state.tabs
    const id = Math.random().toString(16).slice(2, 6)
    tabs.push({
      id,
      name: `Tab ${id}`
    })
    this.setState({ tabs, activeIndex: tabs.length - 1 })
  }
  render() {
    const { activeIndex, tabs } = this.state
    return (
      <Tabs
        dynamic
        activeIndex={activeIndex}
        handleClose={index => this.handleClose(index)}
        onChange={activeIndex => this.setState({ activeIndex })}
      >
        <TabList>
          {tabs.map((item, i) => (
            <Tab key={item.id}>{item.name}</Tab>)
          )}
          <Button ghost icon="add" onClick={e => this.handleAdd(e)} />
        </TabList>
        {tabs.map((item, i) => (
          <TabPanel key={item.id}>{`This is ${item.name}`}</TabPanel>
        ))}
      </Tabs>
    )
  }
}
```
