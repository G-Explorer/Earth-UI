## Basic Usage

```jsx tabs
import { Tooltip, Icon } from '@webapps-ui/core-react'

const TooltipBasic = () => {
  return (
    <Tooltip
      direction="right"
      title="Nullam quis risus eget urna mollisornare vel eu leo. Cum sociis natoquepenatibus et magnis dis parturient montes"
    >
      <Icon type="question" />
    </Tooltip>
  )
}
```

## Trigger by Click

```jsx tabs
import { Tooltip, Icon } from '@webapps-ui/core-react'

const TooltipClick = () => {
  return (
    <Tooltip
      triggerMode="click"
      direction="right"
      title="Nullam quis risus eget urna mollisornare vel eu leo. Cum sociis natoquepenatibus et magnis dis parturient montes"
    >
      <Icon type="question" />
    </Tooltip>
  )
}
```
